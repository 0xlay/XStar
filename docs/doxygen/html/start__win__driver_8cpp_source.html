35 StartWinDriver::StartWinDriver(
36 const wchar_t* serviceName,
37 const wchar_t* pcName,
38 const wchar_t* dbName,
39 uint32_t access
40 )
41 : serviceName_(serviceName), ServiceStatus_{ 0 }
42 {
43 SCM_ = OpenSCManager(pcName, dbName, SC_MANAGER_ALL_ACCESS);
44 if (!SCM_.getHandle())
45 throw WinAPIException();
46
47
48 fullServicePath_.reserve(MAX_PATH);
49 uint32_t cntBytes = GetFullPathNameW(
50 (serviceName_ + L".sys").c_str(),
51 static_cast<DWORD>(fullServicePath_.capacity()),
52 &fullServicePath_[0],
53 nullptr
54 );
55 if (cntBytes == 0)
56 throw WinAPIException();
57
58
59 service_ = OpenService(SCM_.getHandle(), serviceName_.c_str(), access);
60 if (!service_.getHandle())
61 {
62 service_ = CreateService(
63 SCM_.getHandle(),
64 serviceName_.c_str(), serviceName_.c_str(),
65 access | DELETE,
66 SERVICE_KERNEL_DRIVER,
67 SERVICE_DEMAND_START,
68 SERVICE_ERROR_IGNORE,
69 fullServicePath_.c_str(),
70 nullptr, nullptr, nullptr, nullptr, nullptr
71 );
72 if (!service_.getHandle())
73 throw WinAPIException();
74
75
76 serviceCreated_ = true;
77 }
78
79
80 if (!StartService(service_.getHandle(), 0, nullptr))
81 throw WinAPIException();
82 }
